B1. Portfolio generation (2-year, 1000 loans)

Código 
np.random.seed(42); N=1000
tenants = np.random.choice(['FNB','MMC'], N, p=[0.55,0.45])
types = np.random.choice(['Conventional','FHA','VA','USDA'], N, p=[0.40,0.25,0.20,0.15])
amounts = np.clip(85000+np.random.exponential(180000,N), 85000, 1250000).round(-2)
ltv = (75+np.random.beta(5,2,N)*25).round(2)
prop_val = (amounts/(ltv/100)).round(-2)
dti = (18+np.random.beta(3,2,N)*31.5).round(2)
audit_d = np.clip(0.5+np.random.exponential(1.2,N), 0.5, 8.0).round(1)
dates = pd.date_range('2024-01-01','2025-12-31',periods=N)
closing = np.sort(np.random.choice(dates, N, replace=True))

fp = np.zeros(N)
fp += (ltv>93)*0.15 + (dti>43)*0.12 + (np.array(types)=='VA')*0.08
fp += (np.array(types)=='FHA')*0.04 + (audit_d>4)*0.06
fp += np.isin(pd.DatetimeIndex(closing).month,[3,6,9,12])*0.05
fp = np.clip(fp, 0.02, 0.60)
is_fail = (np.random.random(N)<fp).astype(int)

df = pd.DataFrame({'Tenant':tenants,'LoanType':types,'LoanAmount':amounts,
    'PropertyValue':prop_val,'LTV':ltv,'DTI':dti,'AuditDays':audit_d,
    'ClosingDate':closing,'IsFail':is_fail})
df['FY'] = df['ClosingDate'].dt.year
df['Month'] = df['ClosingDate'].dt.month
df['Quarter'] = df['ClosingDate'].dt.quarter
df['IsQtrEnd'] = df['Month'].isin([3,6,9,12]).astype(int)
df['DoW'] = df['ClosingDate'].dt.dayofweek

def penalty(r):
    if r['IsFail']==0: return 0
    b=15000
    if r['LTV']>96.5 and r['LoanType'] in ['FHA','VA']: b+=r['LoanAmount']
    if r['DTI']>45: b+=r['LoanAmount']*0.5
    if r['LoanType']=='VA': b+=50000
    return b
df['Penalty'] = df.apply(penalty, axis=1)

print(f"Portfolio: {N:,} loans | ${df['LoanAmount'].sum()/1e6:.1f}M | Defect: {df['IsFail'].mean()*100:.1f}% | Penalties: ${df['Penalty'].sum()/1e6:.2f}M")

def classify(r):
    if r['LTV']>96.5 and r['LoanType']=='FHA': return 'FHA LTV over limit'
    if r['LoanType']=='VA': return 'VA COE / fee'
    if r['DTI']>43: return 'DTI over limit'
